<script>
(function(){
  const $=(id)=>document.getElementById(id);
  const aEl=$('a'), bEl=$('b'), cEl=$('c');
  const approachEl=$('approach');
  const ans=$('answer');
  const narrative=$('narrative');
  const btnRun=$('btnRun');
  const btnReset=$('btnReset');
  const btnExample=$('btnExample');

  function label(ap){return ap==='a1'?'if–elif with AND':ap==='a2'?'nested if–else':ap==='a3'?'OR shortcut':'iterative updates';}

  function explainPath(ap,a,b,c){
    if(ap==='a1') return (a>=b&&a>=c)?`a≥b and a≥c True → max=a`:`a≥b and a≥c False → ${(b>=c)?'b≥c True → max=b':'b≥c False → max=c'}`;
    if(ap==='a2') return (a>=b)? ((a>=c)?'a≥b True, a≥c True → max=a':'a≥b True, a≥c False → max=c') : ((b>=c)?'a≥b False, b≥c True → max=b':'a≥b False, b≥c False → max=c');
    if(ap==='a3') return (a<b||a<c)? ((b>=c)?'a<b or a<c True, b≥c True → max=b':'a<b or a<c True, b≥c False → max=c') :'a<b or a<c False → max=a';
    if(ap==='a4'){ let mx=a; let s=`max=a`; if(b>mx){mx=b;s+='; b>max True → max=b';} else s+='; b>max False'; if(c>mx){s+='; c>max True → max=c';} else s+='; c>max False'; return s; }
    return '';
  }

  function evaluate(){
    const a=parseFloat(aEl.value)||0, b=parseFloat(bEl.value)||0, c=parseFloat(cEl.value)||0;
    const ap=approachEl.value; let maxv=a;
    if(ap==='a1'){ if(a>=b && a>=c) maxv=a; else if(b>=c) maxv=b; else maxv=c; }
    if(ap==='a2'){ if(a>=b){ if(a>=c) maxv=a; else maxv=c; } else { if(b>=c) maxv=b; else maxv=c; } }
    if(ap==='a3'){ if(a<b||a<c){ if(b>=c) maxv=b; else maxv=c; } else maxv=a; }
    if(ap==='a4'){ maxv=a; if(b>maxv) maxv=b; if(c>maxv) maxv=c; }
    ans.innerHTML=`→ For a=${a}, b=${b}, c=${c} with <b>${label(ap)}</b>, max=${maxv}`;
    narrative.innerHTML=explainPath(ap,a,b,c);
  }

  btnRun.addEventListener('click',evaluate);
  btnReset.addEventListener('click',()=>{aEl.value=1;bEl.value=2;cEl.value=3;approachEl.value='a1';evaluate();});
  btnExample.addEventListener('click',()=>{const samples=[[3,3,3],[5,2,4],[1,9,0],[-2,-5,-1],[10,10,7],[0,-3,-9]];const cur=[parseFloat(aEl.value),parseFloat(bEl.value),parseFloat(cEl.value)];let idx=samples.findIndex(s=>s[0]===cur[0]&&s[1]===cur[1]&&s[2]===cur[2]);idx=(idx+1)%samples.length;[aEl.value,bEl.value,cEl.value]=samples[idx];evaluate();});

  const PROG_KEY='assignProgressFlowMax3';
  const checks=[...document.querySelectorAll('.assign-check')];
  const pText=document.getElementById('progressText');
  const pBar=document.getElementById('progressBar');
  function updateProgress(){const total=checks.length;const done=checks.filter(c=>c.checked).length;pText.textContent=`Completed ${done}/${total}`;pBar.style.width=Math.round((done/total)*100)+'%';}
  checks.forEach(ch=>ch.addEventListener('change',()=>{localStorage.setItem(PROG_KEY,JSON.stringify(checks.map(c=>c.checked?1:0)));updateProgress();}));
  (function(){try{const saved=JSON.parse(localStorage.getItem(PROG_KEY)||'[]');saved.forEach((v,i)=>{if(checks[i])checks[i].checked=!!v;});}catch(e){}updateProgress();})();

  evaluate();
})();
</script>
<script>
(function(){
  const PROG_KEY='assignProgressFlowMax3';
  const checks = Array.from(document.querySelectorAll('.assign-check'));
  const pText = document.getElementById('progressText');
  const pBar = document.getElementById('progressBar');
  function update(){ if(!pText||!pBar) return; const total=checks.length; const done=checks.filter(ch=>ch.checked).length; pText.textContent=`Completed ${done}/${total}`; pBar.style.width = Math.round(done/Math.max(1,total)*100)+'%'; localStorage.setItem(PROG_KEY, JSON.stringify(checks.map(ch=>ch.checked?1:0))); }
  function load(){ try{ const saved=JSON.parse(localStorage.getItem(PROG_KEY)||'[]'); saved.forEach((v,i)=>{ if(checks[i]) checks[i].checked=!!v; }); }catch(e){} update(); }
  checks.forEach(ch=> ch&&ch.addEventListener('change', update));
  load();
})();
</script>
